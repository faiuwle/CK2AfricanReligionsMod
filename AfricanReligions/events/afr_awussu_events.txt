#Carnival of Awussu, pretty lad
character_event = {
	id = AFRREL.035
	desc = EVTDESC_AFRREL035
	picture = GFX_evt_lovers

	capable_only = yes
	prisoner = no
	only_playable = yes

	trigger = {
		OR = {
			religion = amazigh_reformed
			religion = amazigh
		}
		prefers_men_trigger = yes

		OR = {
			NOT = { has_character_flag = fair_event_happened }
			had_character_flag = {
				flag = fair_event_happened
				days = 3
			}
		}
		has_character_modifier = holding_summer_fair
		had_character_flag = {
			flag = summer_fair_begin
			days = 3
		}
		NOT = { has_character_flag = fair_event_lover_lass }	
	}

	immediate = {
		set_character_flag = fair_event_lover_lass
		set_character_flag = fair_event_happened
	}

	mean_time_to_happen = {
		months = 10
	}

	option = {
		name = "EVTOPTA_AFRREL035" #Take a new lover
		create_character = {
			random_traits = yes
			female = no
			dynasty = random
			culture = ROOT
			religion = ROOT
			age = 17			
			trait = fair
		}

		new_character = {
			if = {
				limit = { ROOT = { trait = homosexual } }
				add_trait = homosexual
			}
			else = { remove_trait = homosexual }

			add_lover = ROOT
			remove_trait = genius
			remove_trait = imbecile
			remove_trait = ugly
			remove_trait = slow
			remove_trait = chaste
		}
	}

	option = {
		name = "EVTOPTB_AFRREL035" #Reject the new lover
		random = {
			chance = 25
			add_trait = chaste
		}
	}
}
